GCD LCM 

#include <stdio.h>
 main()
{
int a, b, r, gcd, lcm;
printf("enter two nos.");
scanf("%d%d", &a, &b);
int x=a, y=b;
r=a%b;
while(r!=0)
{
    a=b;
    b=r;
    r=a%b;
}
gcd=b;
printf("gcd= %d\n", gcd);
lcm= (x*y)/gcd;
printf("lcm= %d", lcm);
}

NO. OF DIGITS AND REVERSING THE NO.
#include <stdio.h>
main()  
{  
   int num,count=0 , rev=0, rem;
   printf("enter the no.");
   scanf("%d", &num);
   while(num!=0)
   {
       rem=num%10;
       rev=rev*10+rem;
       num=num/10;
       count++;
   }
   printf("the no. of digits=%d\n", count);
   printf("the reverse of the no. is=%d",rev);
   }  





calculator using switch stmt

#include <stdio.h>
main()  
{  
   int a, b,c, result;

   scanf("%d", &c);
   printf("enter two nos.");
   scanf("%d%d", &a, &b);
   switch(c)
   {
       case 1:result=a+b;

       case 2:result=a-b;
       break;
       case 3:result=a*b;
       break;
       case 4:if(b==0)
       printf("invalid input");
